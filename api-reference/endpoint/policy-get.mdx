---
title: "Get Policy"
api: "GET https://api.magebank.ai/policy/{id}"
description: "Retrieve a specific policy by ID"
---

## Request Headers

<ParamField header="x-api-key" type="string" required>
  Your API key for authentication
</ParamField>

## Path Parameters

<ParamField path="id" type="string" required>
  Policy ID (supports both short IDs like `mage_policy_xxx` and UUIDs)
</ParamField>

## Response

<ResponseField name="id" type="string">
  Policy ID in short format
</ResponseField>

<ResponseField name="type" type="string">
  Policy type (custom, enterprise, developer, default)
</ResponseField>

<ResponseField name="instructions" type="object">
  AI-generated policy instructions
</ResponseField>

<ResponseField name="core_policies" type="object">
  Core policy settings with context
</ResponseField>

<ResponseField name="custom_categories" type="array">
  Custom categories with blocking rules
</ResponseField>

<ResponseField name="enforcement_guidance" type="string">
  Guidance for policy enforcement
</ResponseField>

<ResponseField name="created_at" type="string">
  Creation timestamp
</ResponseField>

<ResponseField name="updated_at" type="string">
  Last update timestamp
</ResponseField>
