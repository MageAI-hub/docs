---
title: "List Policies"
api: "GET https://api.magebank.ai/policy"
description: "List all policies for the authenticated user"
---

## Request Headers

<ParamField header="x-api-key" type="string" required>
  Your API key for authentication
</ParamField>

## Response

<ResponseField name="policies" type="array">
  Array of policy objects
  
  <Expandable title="Policy Object Properties">
    <ResponseField name="id" type="string">
      Policy ID in short format
    </ResponseField>
    
    <ResponseField name="type" type="string">
      Policy type (custom, enterprise, developer, default)
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Creation timestamp
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Last update timestamp
    </ResponseField>
    
    <ResponseField name="has_instructions" type="boolean">
      Whether the policy has AI-generated instructions
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="count" type="number">
  Total number of policies
</ResponseField>
